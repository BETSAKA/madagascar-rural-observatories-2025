hh_size_obs <- res_m_a %>% 
  left_join(res_deb, by = "j5") %>% 
  group_by(j0, j5) %>% 
  mutate(hh_size = n()) %>%

# Calculer la taille moyenne des ménages par hameau
hh_size_obs <- res_m_a %>% 
  left_join(res_deb, by = "j5") %>% 
  group_by(j0, j4, j5) %>% 
  summarise(hh_size = n(), .groups = "drop") %>% 
  mutate(
    j4bis = case_when(
      str_starts(str_to_lower(j4), "bepako") ~ "Bepako",
      str_starts(str_to_lower(j4), "ampijoroa") ~ "Ampijoroa",
      str_starts(str_to_lower(j4), "maroala") ~ "Maroala",
      str_starts(str_to_lower(j4), "madiromiongana") ~ "Madiromiongana",
      TRUE ~ j4
    ),
    Observatory = case_when(
      j0 == 3  ~ "Marovoay",
      j0 == 21 ~ "Alaotra",
      TRUE ~ as.character(j0)
    )
  ) %>% 
  group_by(Observatory, j4bis) %>% 
  summarise(
    n_hh = n(),
    mean_hh_size = mean(hh_size), 
    .groups = "drop"
  )

# Ajouter les lignes de résumé par observatoire
hh_size_combined <- hh_size_obs %>%
  group_by(Observatory) %>%
  summarise(
    j4bis = "Ensemble",
    mean_hh_size = weighted.mean(mean_hh_size, n_hh),
    # On a besoin de n_hh pour pondérer le calcul de la moyenne par observatoire
    n_hh = sum(n_hh) 
  ) %>%
  bind_rows(hh_size_obs, .) %>%
  arrange(Observatory, j4bis == "Ensemble", j4bis)

# Afficher le tableau avec gt

hh_origine <- res_deb %>%

      TRUE    ~ NA_character_

      m3 == 1 ~ "Tompon-tany",

      m3 == 4 ~ "Mpihavy",
      m3 == 5 ~ "Mpandalo (Migrant temporaire)",

    )
  ) %>%
  select(Observatory, j5, Member_Type, Origine)
