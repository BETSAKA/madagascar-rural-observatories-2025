# Calculer la taille moyenne des ménages par hameau
hh_size_obs <- res_m_a %>%

  mutate(
    j4bis = case_when(
      str_starts(str_to_lower(j4), "bepako") ~ "Bepako",
      str_starts(str_to_lower(j4), "ampijoroa") ~ "Ampijoroa",
      str_starts(str_to_lower(j4), "maroala") ~ "Maroala",

      j0 == 3  ~ "Marovoay",

      TRUE ~ as.character(j0)
    )

  summarise(
    n_hh = n(),
    mean_hh_size = mean(hh_size),

hh_age <- res_deb %>%

  mutate(

    Observatory = case_when(

    ),

      m5 <= 10              ~ "0-10 ans",
      between(m5, 11, 20)   ~ "11-20 ans",
      between(m5, 21, 30)   ~ "21-30 ans",
      between(m5, 31, 40)   ~ "31-40 ans",
      between(m5, 41, 50)   ~ "41-50 ans",
      between(m5, 51, 60)   ~ "51-60 ans",
      between(m5, 61, 70)   ~ "61-70 ans",
      m5 >= 71              ~ "71 ans et plus",
      TRUE                  ~ NA_character_
    )
  ) %>%
group_by(Observatory, AgeGroup) %>%
  summarise(n = n(), .groups = "drop") %>%
  group_by(Observatory) %>%
  mutate(Percent = round(n / sum(n) * 100, 1)) %>%
  arrange(Observatory, AgeGroup)

# Ajouter une ligne "Ensemble" par observatoire
hh_age_combined <- hh_age %>%
  group_by(Observatory) %>%
  summarise(
    AgeGroup = "Ensemble",
    n = sum(n),
    Percent = 100,
    .groups = "drop"
  ) %>%
  bind_rows(hh_age, .) %>%
  arrange(Observatory, AgeGroup == "Ensemble", AgeGroup)

# Ligne Total globale
grand_total <- hh_age_combined %>%
  filter(AgeGroup == "Ensemble") %>%
  summarise(
    Observatory = "zzzTotal",   # valeur technique pour le tri
    AgeGroup = "Total",
    n = sum(n),
    Percent = 100
  )

# Fusionner tout
hh_age_final <- bind_rows(hh_age_combined, grand_total) %>%
  mutate(order_flag = ifelse(Observatory == "zzzTotal", 2, 1)) %>%
  arrange(order_flag, Observatory, AgeGroup == "Ensemble", AgeGroup) %>%
  select(-order_flag) %>%
  mutate(Observatory = ifelse(Observatory == "zzzTotal", "", Observatory)) # éviter le groupe vide
