2.2 Description des ménages

Nous allons voir le nombre de ménages enquêtés par observatoire.

Code 
# Tableau de répartition des ménages selon les observatoires
tab <- res_deb %>%
  group_by(j0) %>%
  summarise(n = n_distinct(j5), .groups = "drop") %>%
  arrange(desc(n)) %>%
  mutate(
    Percent = round(n / sum(n) * 100, 1),
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    )
  ) %>%
  select(Observatory, n, Percent)

tab <- tab %>% 
  tibble::add_row(Observatory = "Total", n = sum(tab$n), Percent = sum(tab$Percent, na.rm = TRUE))

# Créer un tableau avec gt 
gt_tab <- gt(tab) %>% 
  tab_header( 
    title = "Répartition des ménages selon les observatoires",
    ) %>% 
  fmt_number( 
    columns = Percent, 
    decimals = 0, 
    suffixing = TRUE ) %>%
  tab_footnote( 
    footnote = "Percent (%)", 
    locations = cells_column_labels(columns = Percent)

2.2.1 Taille des ménages par observatoire

Calculons la taille des ménages par observatoire

Code 
hh_size_obs <- res_m_a %>% 
  left_join(res_deb, by = "j5") %>% 
  group_by(j0, j5) %>% 
  mutate(hh_size = n()) %>% 
  distinct(j0, j5, hh_size) %>% 
  group_by(j0) %>% 
  summarise(mean_hh_size = mean(hh_size), .groups = "drop") %>% 
  mutate(
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    )
  ) %>% 
  select(Observatory, mean_hh_size)

# Displaying with gt
hh_size_obs_tab <- gt(hh_size_obs) %>% 
  tab_header( 
    title = "Taille moyenne des ménages par observatoire",
    ) %>% 
  fmt_number( 
    columns = mean_hh_size, 
    decimals = 2, 
    suffixing = TRUE ) %>%
  cols_label( 
    mean_hh_size = "Taille moyenne" )

2.2.3 Structure par âge

Nous allons voir la répartition des ménages par tranche d’âge par observatoire.

Code 
# Tableau de répartition des ménages par tranche d'âge selon les observatoires

hh_age <- res_deb %>%
  left_join(res_m_a, by="j5") %>% 
  mutate(
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    ),
    AgeGroup = case_when(
      m5 <= 10              ~ "0-10 ans",
      between(m5, 11, 20)   ~ "11-20 ans",
      between(m5, 21, 30)   ~ "21-30 ans",
      between(m5, 31, 40)   ~ "31-40 ans",
      between(m5, 41, 50)   ~ "41-50 ans",
      between(m5, 51, 60)   ~ "51-60 ans",
      between(m5, 61, 70)   ~ "61-70 ans",
      m5 >= 71              ~ "71 ans et plus",
      TRUE                  ~ NA_character_
    )
  ) %>%
  group_by(Observatory, AgeGroup) %>%
  summarise(
    n = n(), 
    .groups = "drop"
  ) %>%
  group_by(Observatory) %>%
  mutate(
    Percent = round(n / sum(n) * 100, 1)
  ) %>%
  arrange(Observatory, AgeGroup)


# Créer un tableau avec gt

    )
