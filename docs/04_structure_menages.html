2.2 Description des ménages

2.2.1 Taille des ménages par observatoire

Calculons la taille des ménages par observatoire

Code 
hh_size_obs <- res_m_a %>% 
  left_join(res_deb, by = "j5") %>% 
  group_by(j0, j5) %>% 
  mutate(hh_size = n()) %>% 
  distinct(j0, j5, hh_size) %>% 
  group_by(j0) %>% 
  summarise(mean_hh_size = mean(hh_size), .groups = "drop") %>% 
  mutate(
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    )
  ) %>% 
  select(Observatory, mean_hh_size)

# Displaying with gt
hh_size_obs_tab <- gt(hh_size_obs) %>% 
  tab_header( 
    title = "Taille moyenne des ménages par observatoire",
    ) %>% 
  fmt_number( 
    columns = mean_hh_size, 
    decimals = 2, 
    suffixing = TRUE ) %>%
  cols_label( 
    mean_hh_size = "Taille moyenne" )

2.2.3 Structure par âge

Nous allons voir la répartition des ménages par tranche d’âge par observatoire.

Code 
# Tableau de répartition des ménages par tranche d'âge selon les observatoires

hh_age <- res_deb %>%
  left_join(res_m_a, by="j5") %>% 
  mutate(
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    ),
    AgeGroup = case_when(
      m5 <= 10              ~ "0-10 ans",
      between(m5, 11, 20)   ~ "11-20 ans",
      between(m5, 21, 30)   ~ "21-30 ans",
      between(m5, 31, 40)   ~ "31-40 ans",
      between(m5, 41, 50)   ~ "41-50 ans",
      between(m5, 51, 60)   ~ "51-60 ans",
      between(m5, 61, 70)   ~ "61-70 ans",
      m5 >= 71              ~ "71 ans et plus",
      TRUE                  ~ NA_character_
    )
  ) %>%
  group_by(Observatory, AgeGroup) %>%
  summarise(
    n = n(), 
    .groups = "drop"
  ) %>%
  group_by(Observatory) %>%
  mutate(
    Percent = round(n / sum(n) * 100, 1)
  ) %>%
  arrange(Observatory, AgeGroup)

2.2.4 Origine

On va voir l’origine des chefs de ménages et de leurs conjoints.

Variables concernées :

m6 : rohim-pihavanana, 1 : CM, 2 : sa femme

m3 : origine

Code 
# Origin of household heads and spouses by observatory

hh_origine <- res_deb %>%
  left_join(res_m_a, by = "j5") %>%
  arrange (j5) %>% 
  filter(m6 %in% c(1, 2)) %>%  # Keep only CM= 1 and spouse = 2
  mutate(
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    ),
    Member_Type = case_when(
      m6 == 1 ~ "Chef de ménage",
      m6 == 2 ~ "Conjoint",
      TRUE    ~ NA_character_
    ),
    Origine = case_when(
      m3 == 1 ~ "Tompon-tany",
      m3 == 2 ~ "Zana-tany",
      m3 == 3 ~ "Valivotaka",
      m3 == 4 ~ "Mpihavy",
      m3 == 5 ~ "Mpandalo (Migrant temporaire)",
      TRUE    ~ NA_character_
    )
  ) %>%
  select(Observatory, j5, Member_Type, Origine)

# Créer le tableau d'analyse par observatoire et origine
origin_table <- hh_origine %>%
  group_by(Observatory, Member_Type, Origine) %>%
  summarise(
    count = n(),
    .groups = 'drop'
  ) %>%
  group_by(Observatory, Member_Type) %>%
  mutate(
    total = sum(count),
    percentage = round((count / total) * 100, 1)
  ) %>%
  ungroup() %>%
  arrange(Observatory, Member_Type, desc(count))

# Tableau GT - Réorganiser les données pour GT
origin_gt_data <- origin_table %>%
  pivot_wider(
    names_from = Member_Type,
    values_from = c(count, percentage),
    names_glue = "{Member_Type}_{.value}",
    values_fill = 0
  ) %>%
  rename(
    `Chef N` = `Chef de ménage_count`,
    `Chef %` = `Chef de ménage_percentage`,
    `Conjoint N` = `Conjoint_count`,
    `Conjoint %` = `Conjoint_percentage`
  ) %>%
  select(Observatory, Origine, `Chef N`, `Chef %`, `Conjoint N`, `Conjoint %`)

# GT tab
gt_origin <- origin_gt_data %>%
  gt(groupname_col = "Observatory") %>%
  tab_header(
    title = "Origine des chefs de ménage et conjoints par observatoire",
    subtitle = "Nombres et pourcentages"
  ) %>%
  cols_label(
    Origine = "Origine",
    `Chef N` = "Nombre",
    `Chef %` = "(%)",
    `Conjoint N` = "Nombre",
    `Conjoint %` = "(%)"
  ) %>%
  tab_spanner(
    label = "Chefs de ménage",
    columns = c(`Chef N`, `Chef %`)
  ) %>%
  tab_spanner(
    label = "Conjoints",
    columns = c(`Conjoint N`, `Conjoint %`)
  ) %>%
  fmt_number(
    columns = c(`Chef N`, `Conjoint N`),
    decimals = 0
  ) %>%
  fmt_number(
    columns = c(`Chef %`, `Conjoint %`),
    decimals = 1,
    suffixing = TRUE
  ) %>%
  tab_style(
    style = cell_fill(color = "#f0f0f0"),
    locations = cells_body(rows = seq(1, nrow(origin_gt_data), 2))
  ) %>%
  tab_footnote(
    footnote = "Les pourcentages sont calculés par catégorie (chef ou conjoint) au sein de chaque observatoire",
    locations = cells_title(groups = "title")
  )
