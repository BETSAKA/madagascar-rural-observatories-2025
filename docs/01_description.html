Il s’agit d’un chapitre préliminaire decrivant le déroulement de l’enquête ainsi que les methodologies déployées.

Cette partie inclut des tables et figures sur :

  • carte des localités enquêtées (avec coordonnées GPS saisies)

  • dénombrement par localité

  • nombre de ménages enquêtés par localité

  • Taux de couverture des localités (nombre dobservations res-deb.dta / dénombrement de la localité)

  • chronogramme des enquêtes (à partir des dates saisies dans les questionnaires)

  • nombre de questionnaire par enquêteur (anonymisés) et par agent de saisie

# Tableau de répartition des ménages selon les observatoires
tab <- res_deb %>%
  group_by(j0) %>%
  summarise(n = n_distinct(j5), .groups = "drop") %>%
  arrange(desc(n)) %>%
  mutate(
    Pourcent = round(n / sum(n) * 100, 1),
    j0 = as.character(j0),
    Observatoire = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    )
  ) %>%
  select(Observatoire, n, Pourcent)

tab <- tab %>% 
  tibble::add_row(Observatoire = "Ensemble", n = sum(tab$n), Pourcent = sum(tab$Pourcent, na.rm = TRUE))

# Créer un tableau avec gt 
gt_tab <- gt(tab) %>% 
  fmt_number( 
    columns = Pourcent, 
    decimals = 0, 
    suffixing = TRUE ) %>%
  tab_footnote( 
    footnote = "Pourcent (%)", 
    locations = cells_column_labels(columns = Pourcent) 
    ) %>% 
  cols_label(
    n = "Effectif"
  ) %>% 
  # Mettre en gras + couleur la ligne des colonnes et "ensemble"

# Calcul des hameaux avec pourcentage interne
tab <- res_deb %>%
  mutate(
    j4bis = case_when(
      str_starts(str_to_lower(j4), "bepako") ~ "Bepako",
      str_starts(str_to_lower(j4), "ampijoroa") ~ "Ampijoroa",
      str_starts(str_to_lower(j4), "maroala") ~ "Maroala",
      str_starts(str_to_lower(j4), "madiromiongana") ~ "Madiromiongana",
      TRUE ~ j4
    ),
    j0 = as.character(j0),
    Observatory = case_when(
      j0 == "3"  ~ "Marovoay",
      j0 == "21" ~ "Alaotra",
      TRUE       ~ j0
    )
  ) %>%
  group_by(Observatory, j4bis) %>%
  summarise(n = n_distinct(j5), .groups = "drop_last") %>%
  mutate(
    Percent = round(n / sum(n) * 100, 1)   # pourcentage interne à chaque observatoire
  ) %>%
  ungroup()

# Totaux par observatoire
totals <- tab %>%
  group_by(Observatory) %>%
  summarise(
    j4bis = "Ensemble",
    n = sum(n),
    Percent = 100
  )

tab <- bind_rows(tab, totals) %>%
  arrange(Observatory, desc(j4bis != "Ensemble"), desc(n))

# Créer le tableau gt avec regroupement par observatoire
gt_tab <- gt(tab, groupname_col = "Observatory") %>%
  tab_header(
    title = "Répartition des ménages par hameau selon les observatoires"
  ) %>%
  fmt_number(
    columns = Percent,
    decimals = 1
  ) %>%
  cols_label(
    j4bis = "Hameau",
    n = "Nombre",
    Percent = "Pourcentage (%)"
  ) %>%
  # Style pour les lignes Totaux
  tab_style(
    style = list(
      cell_text(weight = "bold"),
      cell_fill(color = "#CECECE")   # gris clair
    ),
    locations = cells_body(rows = j4bis == "Ensemble")
  ) %>%
# Style pour les en-têtes de colonnes
  tab_style(
    style = list(
      cell_text(weight = "bold"),
      cell_fill(color = "#4682B4")   # Stellblue
    ),
    locations = cells_column_labels(columns = everything())
  ) %>%
  # Style pour les observatoires
  tab_style(
    style = list(
      cell_text(weight = "bold"),
      cell_fill(color = "#B0E0E6")   # Bleu poudre
    ),
    locations = cells_row_groups()
  ) %>%
  # Style pour le titre principal
  tab_style(
    style = list(
      cell_text(weight = "bold"),
      cell_fill(color = "#FFF2CC")   # beige clair / jaune pâle
    ),
    locations = cells_title(groups = "title")
  )
